```bash
git clone https://github.com/BaekNothing/JsonEditorbyPython.git 
cd JsonEditorbyPython

pip install -r requirments.txt
python main.py
```

or

```bash
git clone https://github.com/BaekNothing/JsonEditorbyPython.git 
cd JsonEditorbyPython

./jsonEditor.exe
```
